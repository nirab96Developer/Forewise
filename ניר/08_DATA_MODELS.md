## User Model — app/models/user.py

```python
class User(Base):
    __tablename__ = "users"
    
    id: int (PK)
    username: str (UNIQUE)
    email: str (UNIQUE)
    full_name: str
    password_hash: str
    
    # Role & Org
    role_id: int (FK → roles)
    region_id: int (FK → regions)
    area_id: int (FK → areas)
    department_id: int (FK → departments)
    manager_id: int (FK → users, self-ref)
    
    # Status
    is_active: bool = True
    two_factor_enabled: bool = False
    last_login: datetime
    
    # Timestamps (BaseModel)
    created_at, updated_at, deleted_at, version
    
    # Relationships
    role → Role
    region → Region
    area → Area
    sessions → [Session]
    otp_tokens → [OTPToken]
    device_tokens → [DeviceToken]
    
# Roles: ADMIN, REGION_MANAGER, AREA_MANAGER, WORK_MANAGER,
#        ACCOUNTANT, ORDER_COORDINATOR, FIELD_WORKER, SUPPLIER, VIEWER
```

## Work Order Model — app/models/work_order.py

```python
class WorkOrder(Base):
    __tablename__ = "work_orders"
    
    id: int (PK)
    order_number: int (UNIQUE)
    title: str
    description: text
    
    # Relations
    project_id: int (FK → projects)
    supplier_id: int (FK → suppliers)
    equipment_id: int (FK → equipment)
    created_by_id: int (FK → users)
    requested_equipment_model_id: int (FK → equipment_models, NOT NULL)
    
    # Status & Priority
    status: str  # PENDING/DISTRIBUTING/APPROVED/ACTIVE/COMPLETED/CANCELLED/REJECTED
    priority: str  # LOW/MEDIUM/HIGH/URGENT
    
    # Work Details
    work_start_date: date
    work_end_date: date
    estimated_hours: decimal
    hourly_rate: decimal
    
    # Financial
    frozen_amount: decimal = 0
    charged_amount: decimal = 0
    
    # Supplier Portal
    portal_token: str (UNIQUE)     ← generated by send-to-supplier
    portal_expiry: datetime         ← now + 3 hours
    portal_token_expires: datetime  ← legacy
    
    # Constraint (Forced Supplier)
    is_forced_selection: bool
    constraint_reason_id: int
    constraint_notes: str
    
    # Timestamps (BaseModel)
    created_at, updated_at, deleted_at, version
    
# STATUS FLOW:
# PENDING → DISTRIBUTING (send-to-supplier)
# DISTRIBUTING → APPROVED (supplier accepts)
# DISTRIBUTING → REJECTED (supplier rejects)
# APPROVED → ACTIVE (start)
# ACTIVE → COMPLETED (close)
# Any → CANCELLED
```

## Project Model — app/models/project.py

```python
class Project(Base):
    __tablename__ = "projects"
    
    id: int (PK)
    code: str (UNIQUE)  # e.g. "YR-001"
    name: str
    description: text
    
    # Organization
    region_id: int (FK → regions)
    area_id: int (FK → areas)
    budget_id: int (FK → budgets)
    manager_id: int (FK → users)
    location_id: int (FK → locations)
    supplier_id: int (FK → suppliers)
    
    # Geography (PostGIS)
    location_geom: geometry(POINT, 4326)  ← GPS point
    forest_id: bigint (FK → forests)
    forest_polygon_id: bigint (FK → forest_polygons)
    
    # Details
    status: str = "active"
    budget: decimal
    start_date: date
    end_date: date
    work_type: str
    execution_type: str
    permit_required: bool = False
    scope: text
    total_hours: decimal
    priority: int
    notes: text
    
    # Relationships
    region → Region
    area → Area
    manager → User
    location → Location
```

```python
class Equipment(Base):
```

## Auth Models

```python
class OTPToken(Base):
    __tablename__ = "otp_tokens"
    id: int
    user_id → users
    token: str          # plain code (legacy)
    code_hash: str      # SHA256 of 6-digit code (new spec)
    token_hash: str     # SHA256 (legacy)
    purpose: login/password_reset
    expires_at: datetime  # now + 5 minutes
    is_used: bool = False
    is_active: bool = True
    attempts: int = 0   # max 3

class DeviceToken(Base):
    __tablename__ = "device_tokens"
    id: UUID
    user_id → users
    token_hash: str (UNIQUE)  # SHA256 of device_token
    device_id: UUID           # client-generated, permanent
    device_name: str          # "Galaxy S23 של נסים"
    device_os: str            # Android/iOS/Windows
    is_active: bool = True
    last_used_at: datetime
    expires_at: datetime      # now + 90 days
    revoked_at: datetime      # set when revoked
    created_at: datetime

class Session(Base):
    __tablename__ = "sessions"
    id: int
    user_id → users
    session_id: str (UNIQUE)
    ip_address: str
    user_agent: str
```

## BaseModel (shared by all models)

```python
class BaseModel(Base):
    """Base class with audit fields"""
    __abstract__ = True
    
    is_active: bool = True
    created_at: datetime = NOW()
    updated_at: datetime = NOW() (auto-update trigger)
    deleted_at: datetime = None  # soft delete
    version: int = 1             # optimistic locking
```

## PostGIS Geometry Models

```python
class Region(Base):
    geom: Geometry(MultiPolygon, SRID=4326)  # גבולות מרחב

class Area(Base):
    geom: Geometry(MultiPolygon, SRID=4326)  # גבולות אזור

class Project(Base):
    location_geom: Geometry(Point, SRID=4326)  # מיקום פרויקט

class Forest(Base):
    geom: Geometry(MultiPolygon, SRID=4326)    # גבולות יער
    area_km2: decimal

class ForestPolygon(Base):
    geom: Geometry(MultiPolygon, SRID=4326)    # פוליגון יער מפורט
    geom_hash: str (UNIQUE)                     # fingerprint

# GEO VALIDATION POLICY (3km rule):
# INSIDE: point within area polygon ✅
# NEAR: 0-3000m from polygon border ⚠️
# FAR: >3000m from polygon ❌ (data issue)
```
